centos：
查看所所有安装的内核版本
rpm -qa| grep kernel

rpm -e 具体内核名和版本

查看所所有安装的内核版本
yum search kernel

yum remove 具体内核名和版本

dpkg --get-selections |grep linux-modules

apt purge linux-modules-?

update-grub

dpkg --purge `dpkg --get-selections | grep deinstall | cut -f1`

apt autoclean
apt clean

yum install yum-utils
package-cleanup --oldkernels --count=1
